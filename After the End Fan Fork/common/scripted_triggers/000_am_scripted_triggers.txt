# Checks if gender laws/culture/religion allows someone to fight
warrior_trigger = {
	OR = {
		AND = {
			is_female = yes
			OR = {
				religion = gaian
				religion = ursuline
				religion = reform
				religion = neomayan
				religion = sol_invicta
				#religion = celtic_pagan #Will be covered by religious doctrines
				#religion = celtic_pagan_reformed
			}
		}
		
		has_character_flag = special_marshal
		
		has_law = true_cognatic_succession
		AND = {
			has_law = agnatic_succession
			is_female = no
		}
		AND = {
			has_law = cognatic_succession
			OR = {
				is_female = no
				martial = 20
			}
		}
		AND = {
			has_law = enatic_succession
			is_female = yes
		}
		AND = {
			has_law = enatic_cognatic_succession
			OR = {
				is_female = yes
				martial = 20
			}
		}
		
		trait = adventurer
		trait = viking
		trait = duelist
		trait = berserker
		trait = peasant_leader
	}
}

is_attracted_to_prev = {
	is_valid_attraction = PREV
	OR = {
		AND = {
			is_female = no
			OR = {
				AND = {
					trait = homosexual
					PREV = { is_female = no }
				}
				PREV = { is_female = yes }
			}
		}
		AND = {
			is_female = yes
			OR = {
				AND = {
					trait = homosexual
					PREV = { is_female = yes }
				}
				PREV = { is_female = no }
			}
		}
	}

	opinion = { who = PREV value = 30 }
}

root_prev_mutual_attraction = {
	ROOT = {
		is_attracted_to_prev = yes
		PREVPREV = {
			is_attracted_to_prev = yes
		}
	}
}
