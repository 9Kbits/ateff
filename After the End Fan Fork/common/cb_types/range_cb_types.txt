liberate_texas = {
	name = CB_NAME_LIBERATE_TEXAS
	war_name = WAR_NAME_LIBERATE_TEXAS
	sprite = 35
	truce_days = 3650
	is_permanent = yes
	check_de_jure_tier = DUKE
	apply_short_occ_mod = no # Do not apply the 'recently_conquered' modifier to Holdings
	allowed_to_target_tributaries = no
	
	sort_priority = 4000
	
	on_add = {
		hidden_tooltip = { fire_haruspicy_event_effect = yes }
		if = {
			limit = { defender = { is_offmap_governor = offmap_china } }
			attacker = { 
				sound_effect = china_angered_emperor
				detract_grace_super_huge_effect = yes
			}
		}
	}
	can_use_gui = {
		ROOT = {
			prestige = 200
		}
	}
	can_use = {
		OR = {
			has_dlc = "Holy Fury" 
			has_dlc = "Jade Dragon"
		}
		ROOT = {
			independent = yes
			OR = {
				has_landed_title = k_westex
				has_landed_title = k_highlands
				has_landed_title = k_yellowrose
				has_landed_title = k_texarkana
				has_landed_title = k_silvercrown
				has_landed_title = k_new_mexico
				has_landed_title = k_rio_grande
				has_landed_title = k_old_star
				has_landed_title = k_comancheria
			}
			culture_group = starfolk
			NOT = {
				has_character_flag = used_texas_liberate_opportunity
			}
			is_nomadic = no
			mercenary = no
			NOT = { pacifist = yes }
			NOT = { same_realm = FROM }
			FROM = {
				NOT = {
					culture_group = starfolk
				}
			}
			FROM = {
				higher_tier_than = DUKE
			}
		}
	}
	
	can_use_title = {
		region = custom_state_texas
	}

	is_valid = {
		ROOT = {
			NOT = { pacifist = yes }
			NOT = { same_realm = FROM }
		}
		FROM = {
			higher_tier_than = DUKE
		}
	}
	
	is_valid_title = {
		FROM = {
			any_realm_province = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}

	on_success = {
		hidden_tooltip = { 
			any_attacker = {
				limit = { NOT = { character = ROOT } }
	
				participation_scaled_prestige = 50
				if = {
					limit = {
						uses_decadence = yes
					}
					participation_scaled_decadence = -10
				}
			}
		}
		
		ROOT = {
			show_scope_change = no
			participation_scaled_prestige = 250
			if = {
				limit = {
					uses_decadence = yes
				}
				participation_scaled_decadence = -10
			}
		}
		FROM = { show_scope_change = no prestige = -50 }
	}

	on_success_title = {
		custom_tooltip = {
			text = texan_liberation_cb_succ_tip
			hidden_tooltip = {
				FROM = {		
					any_realm_province = {
						limit = { 
							duchy = { title = PREVPREV }
							holder_scope = {
								OR = {
									culture = texan
									culture = tejano
									culture = texaner
									culture = czexan
									AND = { 
										culture = comanche
										has_landed_title = k_comancheria
									}
								}
								NOT = { holder_scope = { tier = DUKE } }
								NOT = { holder_scope = { tier = KING } }
							}
						}
						holder_scope = { set_character_flag = future_vassal }
					}			
					any_vassal = {
						limit = { 
							has_character_flag = future_vassal
						}
						set_defacto_liege = ROOT
						clr_character_flag = future_vassal
					}	
				}
				ROOT = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
						#anti_nomad = yes # Vassalize non-nomadic rulers
						type = invasion
					}
				}
			}
		}
	}
	
	

	on_fail = {
		ROOT = { 
			show_scope_change = no
			piety = -125 
			prestige = -250
		}
	}
	
	on_invalidation = {
		ROOT = {
			clr_character_flag = used_texas_liberate_opportunity
		}
	}

	on_reverse_demand = {
		ROOT = {
			show_scope_change = no
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			piety = -250
			prestige = -500
		}
		FROM = {
			show_scope_change = no
			participation_scaled_piety = 200
			participation_scaled_prestige = 100
			if = {
				limit = {
					uses_decadence = yes
				}
				participation_scaled_decadence = -10
			}
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { 
				participation_scaled_piety = 200
				participation_scaled_prestige = 100
				if = {
					limit = {
						uses_decadence = yes
					}
					participation_scaled_decadence = -10
				}
			}
		}
		any_attacker = {
			limit = {
				character = ROOT
				uses_decadence = yes
			}
			participation_scaled_decadence = 20
		}
		any_attacker = {
			limit = {
				NOT = { character = ROOT }
				uses_decadence = yes
			}
			hidden_tooltip = { 
				participation_scaled_decadence = 20
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	ai_will_do = { 
		factor = 1
		modifier = { #Focus on Coronation first.
			factor = 0.1
			has_dlc = "Holy Fury"
			ROOT = {
				is_feudal = yes
				OR = {
					religion = catholic
					religion = fraticelli
				}
				NOR = {
					trait = crowned_by_priest
					trait = crowned_by_bishop
					trait = crowned_by_pope
					trait = crowned_by_myself
				}
			}
		}
	}
}
