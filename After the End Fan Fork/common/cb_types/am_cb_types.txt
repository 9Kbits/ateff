cb_pro_imperial = {

	war_name = WAR_NAME_PRO_IMPERIAL
	sprite = 6
	truce_days = 3650

	can_ask_to_join_war = no
	major_revolt = yes
	
	can_use = {
		ROOT = {
			vassal_of = FROM	
		}
	}

	is_valid_title = {
		NOT = {
			top_liege = { has_landed_title = e_california }
		}
		OR = {
			tier = emperor
			tier = king
			is_primary_holder_title = yes
		}
	}
	
	on_success = {
		FROM = {
			hidden_tooltip = { disband_event_forces = faction_loyalists }
			set_defacto_liege = e_california
		}
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 100
			hidden_tooltip = {
				disband_event_forces = faction_revolters
			}
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = {
				participation_scaled_prestige = 100 
				disband_event_forces = faction_revolters
			}
		}
		e_california = {
			holder_scope = {
				hidden_tooltip = { letter_event = { id = cetic.210 } }
				reverse_imprison = FROM
			}
		}
	}

	#on_success_title = {
	#	succession_w_cooldown = seniority
	#}

	on_fail = {
		ROOT = {
			prestige = -100
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 50
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 50 }
		}
		
		hidden_tooltip = {
			any_attacker = {
				disband_event_forces = faction_revolters
				add_character_modifier = {
					name = faction_succ_seniority_ultimatum_timer
					hidden = yes
					duration = 3650
				}
			}
		}
		FROM = { hidden_tooltip = { disband_event_forces = faction_loyalists } }
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			prisoner = FROM
		}
		FROM = {
			participation_scaled_prestige = 100
			hidden_tooltip = { 
				if = { 
					limit = { NOT = { has_opinion_modifier = { modifier = opinion_traitor who = ROOT } } } #Allow free revocation of a title.

						who = ROOT
						modifier = opinion_traitor
					}
				}
			}
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }

			any_attacker = {
				limit = { liege = { character = FROM } }
				prisoner = FROM
				disband_event_forces = faction_revolters
				add_character_modifier = { #this doesn't exist
					name = faction_pro_imperial_ultimatum_timer

				}

	}

		factor = 100
	}

	}

	}

				port = yes

		}

		if = {
			limit = { defender = { is_offmap_governor = offmap_china } }

				sound_effect = china_angered_emperor

	#	always = yes
	#}
	
	#is_valid_title = {
	#	always = yes
	#}
	
	on_success = {
		ROOT = { 
			prestige = 250
			hidden_tooltip = {
				clr_character_flag = colonist
				#give_nickname = nick_the_conqueror
				if = {
					limit = { has_landed_title = e_edokko }
					narrative_event = { id = col.12 days = 1 } # Victorious
				}
				if = {
					limit = { has_landed_title = e_novorusskiy }
					narrative_event = { id = col.22 days = 1 } # Victorious
				}
			}
		}
		FROM = {
			prestige = -250 
		}
	}

	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				FROM = {
					random_realm_province = {
						limit = {
							port = yes
							de_jure_liege_or_above = PREVPREV
							any_province_holding = { holding_type = city }
							#OR = {
							#	NOT = { culture = ROOT }
							#	NOT = { religion = ROOT }
							#}
						}
						culture = ROOT
						religion = ROOT
						
						# Make it a merchant republic
						random_province_holding = {
							limit = { holding_type = city }
							province_capital = yes
							grant_title = ROOT
						}
						county = { grant_title = ROOT }
						duchy = { grant_title = ROOT }
						# Give them the old capital too
						any_province_holding = {
							limit = { same_realm = FROM }
							grant_title = ROOT
						}
					}
				}
				ROOT = {
					set_government_type = colonial_government
					occupy_minors_of_occupied_settlements = FROM
					gain_all_occupied_titles = FROM
					vassalize_or_take_under_title_destroy_duchies = {
						title = PREV
						enemy = FROM
						type = invasion
					}
					
					# Settle the colonists
					any_demesne_title = {
						limit = { tier = BARON }
						remove_holding_modifier = recently_conquered
						refill_holding_levy = yes
					}
				}
			}
		}
		
	}

	on_fail = {
		FROM = {
			prestige = 100
		}
		ROOT = {
			imprison = FROM 
			hidden_tooltip = {
				disband_event_forces = yes
				clr_character_flag = colonist
			}
		}
	
		ROOT = { prestige = -100 }

			limit = { ROOT = { has_landed_title = e_edokko } }

			character_event = { id = col.1 }
			narrative_event = { id = col.23 days = 1 } # Defeated
		}
	}

		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		
		FROM = {
			prestige = 200
		}
		
		ROOT = {
			imprison = FROM
			prestige = -200
			hidden_tooltip = {
				disband_event_forces = yes
				clr_character_flag = colonist
			}
		}
		
		random = {
			chance = 40
			if = {
				limit = { ROOT = { has_landed_title = e_edokko } }
				narrative_event = { id = col.13 days = 1 } # Defeated
			}
			if = {
				limit = { ROOT = { has_landed_title = e_novorusskiy } }

		}
		if = {
			limit = { ROOT = { has_landed_title = e_novorusskiy } }

		}
	}

				clr_character_flag = colonist
			}
		}

			chance = 40
			if = {
				limit = { ROOT = { has_landed_title = e_edokko } }
				narrative_event = { id = col.13 days = 1 } # Defeated
			}
			if = {
				limit = { ROOT = { has_landed_title = e_novorusskiy } }

			}
		}
		if = {

		if = {

			prisoner = no
			is_adult = yes
			OR = {
				NOT = { has_character_modifier = holy_war_cooldown }

			}
			custom_tooltip = {
				text = RECONQUISTAS_STARTED

		ROOT = {
			has_character_flag = currently_reconquering_mexico

			any_realm_province = {
				OR = {
					AND = {
						any_neighbor_province = {
							region = custom_jalisco
						}
					}
					AND = {
						NOT = {
							any_neighbor_province = {
								region = custom_jalisco
							}
						}
						NOT = {
							any_neighbor_province = {
								region = custom_empire_mexico

										AND = {
											independent = yes

										}

											NOT = {
												religion_group = cult_of_saints
											}
										}
									}
								}
							}
						}
						any_neighbor_province = {
							is_land = yes
							any_neighbor_province = {
								region = custom_jalisco
							}
						}
					}

						NOR = {
							any_neighbor_province = {
								region = custom_jalisco
							}
							any_neighbor_province = {
								is_land = yes
								any_neighbor_province = {
									region = custom_jalisco
								}
							}
						}
						NOT = {
							any_neighbor_province = {
								is_land = yes

									region = custom_empire_mexico

											AND = {

											}

												NOT = {
													religion_group = cult_of_saints
												}
											}
										}
									}
								}
							}
						}

							is_land = yes
							any_neighbor_province = {
								is_land = yes
								any_neighbor_province = {
									region = custom_jalisco
								}
							}
						}
					}
				}
			}

			conditional_tooltip = {
				trigger = {
					ai = yes
				}

				NOT = {
					has_character_modifier = reconquista_preparation_duration
				}
			}
		}

			any_realm_province = {
				region = custom_jalisco
			}
			NOT = {
				religion_group = ROOT
			}
		}
	}

			}

			text = reconquista_valid_2_TT

				NOT = {
					religion_group = ROOT
				}
			}
