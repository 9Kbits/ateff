holy_order_distribute_titles_effect = {
	any_demesne_title = {
		limit = {
			NOT = {
				holding_type = castle
			}
			lower_tier_than = DUKE
		}
		trigger_switch = {
			on_trigger = holding_type
			city = {
				create_random_steward = {
					random_traits = yes
					dynasty = random
				}
			}
			temple = {
				create_random_priest = {
					random_traits = yes
					dynasty = random
				}
			}
			fallback = {
				create_random_soldier = {
					random_traits = yes
					dynasty = random
				}
			}
		}
		grant_title_no_opinion = new_character
	}
