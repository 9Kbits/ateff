			}

				set_description = "bloodline_desc_libertador"

			}
			new_bloodline = {
				set_description = "bloodline_desc_libertador"
			}	
			sound_effect = bloodline_added		
		}
	}
	option = {
		name = EVTOPTB_patria.2
		trigger = {
			NOT = { has_landed_title = e_gran_colombia }

		}

	option = {
		name = EVTOPTC_patria.2
		trigger = {

			}

	}
