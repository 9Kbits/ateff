# Confirm decision and check religion

character_event = {
	id = SoA.5000
	desc = EVTDESC_SoA_5000
	picture = GFX_evt_bishop
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes

	option = {
		name = EVTOPTA_SoA_5000
		piety = 20
		set_character_flag = preparing_pilgrimage
		clr_character_flag = activated_pilgrimage
		hidden_tooltip = {
			if = {
				limit = {
					OR = {
						religion = evangelical  # Washington, Atlanta, Philly
						religion = antinomian
						religion = charismatic
						religion = postadventist
					}
				}
				character_event = { id = SoA.5011 }
			}
			if = {
				limit = {
					OR = {
						religion = anabaptist # Washington, Philly, Pittsburgh
						religion = batenburger
					}
				}
				character_event = { id = SoA.5011 }
			}
			if = {
				limit = {
					OR = {
						religion = catholic # All Catholic holy places
						religion = neognostic
						religion = sedevacantist
						religion = nestorian
					}
				}
				character_event = { id = SoA.5008 }
			}
			if = {
				limit = {
					OR = {
						religion = ursuline
						religion = angeline
						religion = high_church
						religion = united_church
						religion = confederated
					}
				}
				character_event = { id = canada.15 }
			}
			if = {
				limit = {
					OR = {
						religion = falling_star
						religion = moravian
					}
				}
				character_event = { id = fallingstar.1 }
			}
		}
	}
}

# Check catholic pilgrim's location - then send on to destination choice event
character_event = {
	id = SoA.5008

	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		hidden_tooltip = {
			if = {
				limit = {
					OR = {
						capital_scope = { province_id = 389 } #Denver
						capital_scope = { province_id = 170 } #Chicago
						capital_scope = { province_id = 1050 } #St. Louis
						capital_scope = { province_id = 1231 } #Bexar
					}
				}
				character_event = { id = SoA.5016 days = 4 } # Go to Washington
			}
			if = {
				limit = {
					capital_scope = { province_id = 63 } #Washington
					NOT = { religion = sedevacantist }
				}
				character_event = { id = SoA.5017 days = 4 } # Go to Rome
			}
			if = {
				limit = {
					capital_scope = { province_id = 63 } #Washington
					religion = sedevacantist
				}
				character_event = { id = SoA.5018 days = 4 } # May choose destination
			}
			if = {
				limit = { 
					NOT = {
						OR = {
							capital_scope = { province_id = 389 } #Denver
							AND = {
								capital_scope = { province_id = 63 } #Washington
								religion = sedevacantist
							}
							capital_scope = { province_id = 170 } #Chicago
							capital_scope = { province_id = 1050 } #St. Louis
							capital_scope = { province_id = 1231 } #Bexar
						}
					}
				}
				character_event = { id = SoA.5018 days = 4 } # May choose destination
			}
		}
	}
}
