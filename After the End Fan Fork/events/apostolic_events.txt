# Follower of Bonfim may gain Sympathy for Afro-Syncretic
character_event = {
	id = apostolic.2
	desc = EVTDESC_apostolic_2
	picture = GFX_evt_candomble
	border = GFX_event_normal_frame_religion
	
	trigger = {
		religion = apostolic
		trait = apostolic_bonfim
		NOT = { trait = sympathy_afro_syncretic }
	}
	
	mean_time_to_happen = {
		months = 40
	}
	
	option = {
		name = EVTOPTA_apostolic_2
		remove_trait = zealous
		add_trait = sympathy_afro_syncretic
	}
	
	option = {
		name = EVTOPTB_apostolic_2
		piety = 25
	}
}

# Follower of Pedro strives to be more like him
character_event = {
	id = apostolic.4
	desc = EVTDESC_apostolic_4
	picture = GFX_evt_lady_guadalupe
	border = GFX_event_normal_frame_religion
	
	trigger = {
		religion = apostolic
		trait = apostolic_pedro
		NOT = {
			trait = cynical
		}
		NOT = {
			AND = {
				trait = diligent
				trait = humble
				trait = brave
			}
		}
		NOT = {
			has_character_flag = like_pedro
		}
	}
	
	immediate = {
		set_character_flag = like_pedro
	}
	
	mean_time_to_happen = {
		months = 40
	}
	
	option = {
		name = EVTOPTA_apostolic_4
		trigger = {
			NOT = {
				trait = diligent
			}
		}
		add_trait = diligent
	}
	
	option = {
		name = EVTOPTB_apostolic_3
		trigger = {
			NOT = {
				trait = humble
			}
		}
		add_trait = humble
	}
	
	option = {
		name = EVTOPTC_apostolic_3
		trigger = {
			NOT = {
				trait = brave
			}
		}
		add_trait = brave
	}
	
	option = {
		name = EVTOPTD_apostolic_3
	}
}
