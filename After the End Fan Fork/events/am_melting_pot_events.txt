#Melting pot events for the new AtE melting pot cultures
namespace = meltingpot

# The New English Melting Pot
character_event = {
	id = meltingpot.1
	desc = "EVTDESC_melting_pot_1"
	picture = GFX_evt_stone_church
	
	only_playable = yes
	culture = british
	
	trigger = {
		year = 2900
		any_demesne_province = {
			region = custom_new_england
			culture = yankee
			any_province_holding = {
				NOT = { holding_type = nomad }
			}
		}
		#NOT = { government = chinese_imperial_government }
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.75
			learning = 7
		}
		modifier = {
			factor = 0.75
			learning = 9
		}
		modifier = {
			factor = 0.75
			learning = 11
		}
		modifier = {
			factor = 0.75
			learning = 13
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 3 }
		}
		modifier = {
			factor = 2.0
			NOT = { learning = 1 }
		}
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = "EVTOPTA_melting_pot_1"
		culture = new_english
		random_demesne_province = {
			limit = {
				region = custom_new_england
				culture = yankee
				any_province_holding = {
					NOT = { holding_type = nomad }
				}
			}
			culture = new_english
		}
	}
}

#Make my court speak English
character_event = {
	id = meltingpot.3
	desc = "EVTDESC_melting_pot_3"
	picture = GFX_evt_throne_room
	
	only_rulers = yes
	culture = new_english

	trigger = {
		year = 1100
		top_liege = {
			primary_title = {
				OR = {
					title = k_new_england
					title = e_redcoat_dominion
				}
			}
		}
		any_courtier = {
			OR = {
				culture = yankee
				culture = british
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = "EVTOPTA_melting_pot_3"
		random_courtier = {
			limit = {
				OR = {	
					culture = yankee
					culture = british
				}
			}
			culture = new_english
			hidden_tooltip = {
				character_event = { id = meltingpot.5 days = 3 }
			}
		}
	}
}

# Province adopts Carpetbagger culture
province_event = {
	id = meltingpot.6
	desc = EVTDESC_melting_pot_6
	picture = GFX_evt_stone_church
	border = GFX_event_normal_frame_diplomacy
	
	trigger = {
		OR = {
			culture = dixie
			culture = appalachian
			culture = melungeon
		}
		owner = {
			OR = {
				culture = yankee
				culture = carpetbagger
			}
		}
		NOT = { year = 2750 }
	}
	
	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTA_melting_pot_6
		trigger = {
			NOT = {
				owner = { culture = carpetbagger }
			}
		}
		culture = carpetbagger
	}
	option = {
		name = EVTOPTB_melting_pot_6
		trigger = {
			owner = { culture = carpetbagger }
		}
		culture = carpetbagger
	}
}
