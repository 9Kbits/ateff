# Japan - Player Controlled
character_event = {
	id = col.40
	desc = col.11.text
	picture = GFX_evt_colonial_japan
	hide_window = yes
	
	trigger = {
		AND = {
			ai = no
			tier = duke
			religion = shinto
			culture = edokko
			capital_scope = {
				region = custom_pacific
			}
		}
		NOT = { 
			has_game_rule = {
				name = colonisation_invasions
				value = off
			}
		}
		NOT = { is_title_active = e_edokko }
		NOT = { has_global_flag = edokko_defeated }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		activate_title = { title = e_edokko status = yes }
		wealth = 5000
		e_edokko = {
			grant_title = ROOT
		}
		add_artifact = ate_firearm
		set_character_flag = colonist
		narrative_event = { id = col.41 }
		capital_scope = {
			random_province_holding = {
				limit = {
					holding_type = city
					holder_scope = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
						}
					}
				}
				if = {
					limit = {
						NOT = { holder = ROOT }
					}
					usurp_title = ROOT
				}
				make_capital_holding = yes
			}
			create_character = {
				random_traits = yes
				dynasty = none
				religion = shinto
				culture = edokko
				female = no
				age = 30
			}
			new_character = {
				add_artifact = ate_firearm
				character_event = { id = salvage.140 } # Give random salvage
				character_event = { id = misc.3 } # Diversity
				character_event = { id = misc.4 } # Physique
				spawn_unit = {
					province = PREV
						troops = {
						horse_archers = { 1125 1125 }
						light_infantry = { 900 900 }
						heavy_infantry = { 1125 1125 }
						pikemen = { 900 900 }
						archers = { 375 375 }
					}
					attrition = 0.25
				}
			}
		}

	}
}
