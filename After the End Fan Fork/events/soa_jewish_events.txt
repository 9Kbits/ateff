# Jewish Diplomat Appears at Court (Yiddish)
character_event = {
	id = SoA.200
	desc = EVTDESC_SoA_200
	picture = GFX_evt_emissary
	border = GFX_event_normal_frame_religion
	
	min_age = 16
	prisoner = no
	only_playable = yes
	capable_only = yes
	only_independent = yes
	
	trigger = {
		has_dlc = "Sons of Abraham"
		OR = {
			religion_group = christian
			religion_group = cult_of_saints
			religion_group = oldworld
			religion_group = latter_day_saints
			religion_group = jewish_group
			religion_group = muslim
			AND = {
				religion = rastafarian
				trait = twelve_tribes
			}
		}
		NOR = { 
			capital_scope = {
				region = custom_neoladino
			}
			capital_scope = {
				region = world_caribbean
			}
		}
		NOT = { has_character_modifier = expelled_jewish }
		NOT = { trait = incapable }
		NOT = { is_inaccessible_trigger = yes }
	}
	
	mean_time_to_happen = {
		months = 1100
	}
	
	immediate = {
		create_character = {
			random_traits = yes
			culture = yiddish
			dynasty = culture
			religion = jewish
			female = no
			age = 30
			trait = charismatic_negotiator
			attributes = {
				diplomacy = 10
			}
			flag = ai_flag_refuse_conversion
		}
		new_character = {
			character_event = { id = misc.3 } # Diversity
			character_event = { id = misc.4 } # Physique
			remove_trait = socializer
			remove_trait = gamer
			remove_trait = hedonist
			remove_trait = duelist
			remove_trait = hunter
			remove_trait = strategist
			remove_trait = administrator
			remove_trait = architect
			remove_trait = gardener
			remove_trait = schemer
			remove_trait = impaler
			remove_trait = seductress
			remove_trait = scholar
			remove_trait = theologian
			remove_trait = mystic
			random_list = {
				70 = {
					add_trait = socializer
				}
				30 = {
					add_trait = socializer
					add_trait = shrewd
				}
			}
			save_event_target_as = jewish_courtier
		}
	}
	
	option = {
		name = EVTOPTA_SoA_200
		custom_tooltip = {
			text = jewish_courtier_tooltip
		}
	}
}
