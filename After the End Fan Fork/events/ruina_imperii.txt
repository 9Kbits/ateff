namespace = ruin

# Fall of generic empire
narrative_event = {
	id = ruin.1
	title = TITLE_collapse
	desc = EVTDESC_collapse
	picture = GFX_evt_council
	
	major = yes	
	only_playable = yes

	trigger = {
		independent = yes
		real_tier = emperor
		NOT = {
			government = nomadic_government
		}
		OR = {
			AND = {
				NOT = {
					primary_title = {
						num_title_realm_provs = { 
							who = ROOT
							value = 100
						}	
					}
				}				
				NOT = {
					realm_size = 100
				}			
			}
			AND = {				
				NOT = {
					realm_size = 70
				}	
				NOT = {
					dynastic_prestige = 1500
				}
			}
		}
		AND = {
			NOT = {
				primary_title = {
					title = e_redcoat_dominion
				}		
			}
			NOT = {
				primary_title = {
					title = e_brazil
				}

			}
		}
		
		primary_title = {
			is_titular = yes
			temporary = no
		}
	
	}
	mean_time_to_happen = {
		months = 20
	}	
	after = {
		FROM = {
			character_event = { id = ruin.3 }
			# primary_title = {
				# destroy_landed_title = yes
			# }
			# prestige = -500
		}
	}
	option = {
		name = "RUIN_generic"
		trigger = {
			always = yes
		}		
	}
}
