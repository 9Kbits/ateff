#Fall of Caribbean empire
narrative_event = {
	id = ruin.2
	title = TITLE_collapse_CARIBBEAN
	desc = EVTDESC_collapse_CARIBBEAN
	picture = GFX_evt_council	

	major = yes
	only_playable = yes
	hide_from = yes

	trigger = {
		NOT = {
			has_game_rule = {
				name = caribbean_collapse
				value = off
			}
		}
		independent = yes
		real_tier = emperor
		NOT = {
			has_character_flag = empress_portia
		}
		NOT = {
			year = 2710
		}
		OR = {
			AND = {				
				NOT = {
					realm_size = 90
				}
			}
			AND = {
				NOT = {
					completely_controls = d_haiti
				}
				NOT = {
					completely_controls = d_dominica
				}
				NOT = {
					completely_controls = k_west_indies
				}
				NOT = {
					completely_controls = k_cuba
				}
			}
		}
		NOT = {
			realm_stewardship = 26
		}
		NOT = {
			realm_diplomacy = 26
		}
		primary_title = {
			temporary = no
			title = e_caribbean
		}
	}	
	mean_time_to_happen = {
		months = 25
	}
	immediate = {
		primary_title = {
			destroy_landed_title = yes
		}
		prestige = -500
	}	
	option = {
		name = "RUIN_CARIBBEAN_NORM"
		trigger = {
			always = yes
		}	
	}
}
