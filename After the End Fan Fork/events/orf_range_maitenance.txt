namespace = ORFmaint

#events to preserve orf status
#startup
character_event = {
	id = ORFmaint.11
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		is_feudal = yes
		is_landed = yes
		is_save_game = no
		NOT = { has_character_flag = ORF_embrace }
		NOT = { has_character_flag = ORF_reject }
		independent = yes
	}

	immediate = {
		if = {
			limit = {
				OR = {
					capital_scope = {
						region = custom_range_range
					} 
					has_character_flag = ORF_embrace
					culture = altiplanense
					culture = bravo
					culture = neomexicano
					culture_group = starfolk
					culture = arixan
					culture = comanche
					culture = sioux
				}			
			}
			any_realm_character = {
				limit = { 
					is_landed = yes
					is_feudal = yes
				}
				clr_character_flag = ORF_reject
				set_character_flag = ORF_embrace
				set_government_type = open_range_feudalism_government
			}
			set_character_flag = ORF_embrace
		}	
		else = {
			set_character_flag = ORF_reject
			any_realm_character = {
				limit = { 
					is_landed = yes
					is_feudal = yes
				}
				clr_character_flag = ORF_embrace
				set_character_flag = ORF_reject
				set_government_type = feudal_government
			}
		}	
		set_character_flag = startup_orf_done
	}
}

#new holder
character_event = {
	id = ORFmaint.12
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		OR = {
			has_character_flag = ORF_reject
			has_character_flag = ORF_embrace
		}
		is_feudal = yes
	}

	immediate = {
		if = {
			limit = {
				has_character_flag = ORF_embrace
			}
			any_demesne_title = {
				set_title_flag = ORF_embrace
			}
			
		}	
		else_if = {
			limit = {
				FROMFROM = { has_character_flag = ORF_reject }		
			}
			set_character_flag = ORF_reject
			limit = {
				has_character_flag = ORF_reject
			}
			any_demesne_title = {
				set_title_flag = ORF_reject
			}
		}
	}
}
