narrative_event = { #Louisiana Integrated into HCC
	id = hcc.1
	title = EVTTITLE_hcc_1
	desc = EVTDESC_hcc_1
	picture = GFX_evt_holy_emperor
	is_triggered_only = yes
	
	major = yes	
	only_playable = yes
	
	trigger = {
		is_playable = yes
		capital_scope = { region = custom_usa }
		has_global_flag = louisiana_integrated
	}
	
	option = {
		name = EVTOPTA_hcc_1
		trigger = {
			has_landed_title = e_dixie
		}
		prestige = 1000
	}
	
	option = {
		name = EVTOPTB_hcc_1
		trigger = {
			NOT = { has_landed_title = e_dixie }
			top_liege = {
				has_landed_title = e_dixie
			}
		}
	}
	
	option = {
		name = EVTOPTC_hcc_1
		trigger = {
			NOR = {
				has_landed_title = e_dixie
				top_liege = {
					has_landed_title = e_dixie
				}
			}
			culture_group = south_anglo
		}
	}
	
	option = {
		name = EVTOPTD_hcc_1
		trigger = {
			NOR = {
				has_landed_title = e_dixie
				top_liege = {
					has_landed_title = e_dixie
				}
			}
			NOT = { culture_group = south_anglo }
		}
	}
}
