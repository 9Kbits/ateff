# Knights Hospitaller asks to receive an island 
letter_event = {
	id = HFP.44001 
	desc = EVTDESC_HFP_44001
	is_triggered_only = yes

	option = {
		name = EVTOPTA_HFP_44001
		piety = 100
		scaled_wealth = { value = 2 }

		d_knights_of_peter_claver = {
			show_scope_change = no
			holder_scope = {
				show_scope_change = no
				opinion = {
					who = ROOT
					modifier = opinion_grateful
					years = 10
				}
				opinion = {
					modifier = in_non_aggression_pact
					years = 10
					who = ROOT
				}
			}
		}

		custom_tooltip = {
			text = EVTOPTA_HFP_44001_TT

			event_target:potential_title = {
				if = {
					limit = {
						owner = {
							OR = {
								character = ROOT
								any_liege = {
									character = ROOT
								}
							}
						}
						NOT = {
							tier = DUKE
						}
					}
					grant_title_no_opinion = FROM
				}
				else_if = {
					limit = {
						owner = {
							OR = {
								character = ROOT
								any_liege = {
									character = ROOT
								}
							}
							tier = DUKE
						}
					}
					destroy_landed_title = THIS
				}
				any_direct_de_jure_vassal_title = {
					limit = {
						OR = {
							owner = {
								character = ROOT
							}
							owner = {
								any_liege = {
									character = ROOT
								}
							}
						}
					}
					grant_title_no_opinion = FROM
					any_direct_de_jure_vassal_title = {
						limit = {
							OR = {
								owner = {
									character = ROOT
								}
								owner = {
									any_liege = {
										character = ROOT
									}
								}
							}
						}
						grant_title_no_opinion = FROM
					}
				}
			}
			FROM = {
				any_demesne_title = {
					limit = {
						NOR = {
							AND = {
								OR = {
									tier = BARON
									tier = COUNT
								}
								holding_type = CASTLE
							}
							title = d_knights_of_peter_claver
						}
					}
					if = {
						limit = {
							tier = BARON
							holding_type = CITY
						}
						create_random_steward = {
							random_traits = yes
							dynasty = random
						}
					}
					else_if = {
						limit = {
							tier = BARON
							holding_type = TEMPLE
						}
						create_random_priest = {
							random_traits = yes
							dynasty = random
						}
					}
					else = {
						create_random_soldier = {
							random_traits = yes
							dynasty = random
						}
					}
					grant_title_no_opinion = new_character
				}
			}

			d_knights_of_peter_claver = {
				set_name = "knights_hospitaler_island_rename"
			}
			event_target:potential_title = {
				capital_scope = {
					FROM = {
						if = {
							limit = {
								has_landed_title = PREV
								event_target:potential_title = {
									tier = DUKE
								}
							}
							capital = PREV
						}
						else = {
							capital = event_target:potential_title
						}
					}
				}
			}
			any_playable_ruler = {
				limit = {
					religion = FROM
				}
				narrative_event = { id = HFP.44004 }
			}
			ROOT = {
				narrative_event = { id = HFP.44004}
			}
			FROM = {
				character_event = { id = HFP.41090 days = 1 }
				character_event = { id = HFP.41090 days = 3 }
			}
		}

		ai_chance = {
			factor = 1
		}
	}

	option = {
		name = EVTOPTB_HFP_44001
		piety = -25

		set_character_flag = refused_to_give_island_to_knights

		ai_chance = {
			factor = 0
		}
	}
}

# Holy Order decides their liege is unfit to rule over them
character_event = {
	id = HFP.44002
	hide_window = yes
	is_triggered_only = yes
	ai = yes

	trigger = {
		holy_order = yes
		independent = no
	}

	immediate = {
		liege = {
			save_event_target_as = liege_character
		}

		if = {
			limit = {
				OR = {
					NOT = {
						opinion = {
							who = liege
							value = -20
						}
					}
					liege = {
						excommunicated_for = ROOT
					}
					liege = {
						NOT = {
							religion = ROOT
						}
					}
				}
			}
			opinion = {
				who = event_target:liege_character
				modifier = opinion_unfit_ruler
				years = 30
			}
			liege = {
				letter_event = { id = HFP.44003 }
			}
			any_demesne_title = {
				limit = {
					holy_order = no
					OR = {
						county = {
							owner = {
								OR = {
									character = event_target:liege_character
									event_target:liege_character = {
										any_vassal = {
											NOT = {
												character = ROOT
											}
											character = PREVPREV
										}
									}
								}
							}
						}
						duchy = {
							owner = {
								OR = {
									character = event_target:liege_character
									event_target:liege_character = {
										any_vassal = {
											NOT = {
												character = ROOT
											}
											character = PREVPREV
										}
									}
								}
							}
						}
						kingdom = {
							owner = {
								OR = {
									character = event_target:liege_character
									event_target:liege_character = {
										any_vassal = {
											NOT = {
												character = ROOT
											}
											character = PREVPREV
										}
									}
								}
							}
						}
						empire = {
							owner = {
								OR = {
									character = event_target:liege_character
									event_target:liege_character = {
										any_vassal = {
											not = {
												character = ROOT
											}
											character = PREVPREV
										}
									}
								}
							}
						}
					}
				}
				event_target:liege_character = {
					grant_title_no_opinion = PREV
				}
			}
			set_defacto_liege = ROOT
		}
	}
}

	option = {
		name = EVTOPTA_HFP_44006

		piety = 100
		scaled_wealth = { value = 2 }

			show_scope_change = no
			holder_scope = {
				show_scope_change = no
				opinion = {
					who = ROOT
					modifier = opinion_grateful
					years = 10
				}
				opinion = {
					modifier = in_non_aggression_pact
					years = 10
					who = ROOT
				}
			}
		}

		custom_tooltip = {
			text = EVTOPTA_HFP_44006_TT

			event_target:holy_order_potential_title = {
				# Destroy the Duchy title, if there is any
				if = {
					limit = {
						owner = {
							OR = {
								character = ROOT
								any_liege = {
									character = ROOT
								}
							}
						}
					}
					destroy_landed_title = THIS
				}
				# Give the Teutonics all the land you hold in the area
				any_direct_de_jure_vassal_title = {
					limit = {
						owner = {
							OR = {
								character = ROOT
								any_liege = {
									character = ROOT
								}
							}
						}
					}
					grant_title_no_opinion = FROM
					any_direct_de_jure_vassal_title = {
						limit = {
							owner = {
								OR = {
									character = ROOT
									any_liege = {
										character = ROOT
									}
								}
							}
						}	
						grant_title_no_opinion = FROM
					}
				}
			}
			# Upgrade tribal areas into proper feudal areas
			FROM = {
				teutonic_order_upgrade_capital_effect = yes
			}
			# Give out any unfit land (see cities, temples) to new vassals
			FROM = {
				holy_order_distribute_titles_effect = yes
			}
			# Refilled holdings and remove province penalties
			FROM = {
				character_event = { id = HFP.41090 days = 1 }
				character_event = { id = HFP.41090 days = 3 }
			}
			# Making sure they Teutonics stay proper Holy Orderish
			FROM = {
				set_government_type = order_government
			}
			# Sending out narrative event to all Catholic rulers
			if = {
				limit = {
					FROM = {
						any_demesne_title = {
							tier = COUNT
						}
					}
				}
				any_playable_ruler = {
					limit = {
						religion = FROM
					}
					narrative_event = { id = HFP.44007 }
				}
				ROOT = {
					narrative_event = { id = HFP.44007 }
				}
			}
		}
		ai_chance = {
			factor = 1
		}
	}
	option = {
		name = EVTOPTB_HFP_44006

		set_character_flag = refused_to_give_land_to_ursulines
		ai_chance = {
			factor = 0
		}
	}
	after = {
		clr_global_flag = ursulines_asked_for_land
	}
}
