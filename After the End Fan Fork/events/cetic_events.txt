character_event = {
	id = cetic.10
	desc = EVTDESC_bounce
	picture = GFX_evt_chinese_bureaucrat
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		OR = {
			has_landed_title = e_california
			has_landed_title = e_new_california
		}
	}
	
	immediate = {
		current_heir = { 
			add_character_modifier = {
				name = imperial_authority
				duration = -1
			}
			if = {
				limit = {
					check_variable = { which = global_imp_auth value < 11 }
				}
				add_trait = weak_emperor
				add_trait = figurehead
			}
			if = {
				limit = {
					check_variable = { which = global_imp_auth value <= 50 }
				}
				add_trait = figurehead
				character_event = { id = cetic.316 }
			}
			if = {
				limit = {
					AND = {
						check_variable = { which = global_imp_auth value = 50 }
						check_variable = { which = global_imp_auth value < 70 }
					}
				}
				character_event = { id = cetic.316 }
			}
			if = {
				limit = {
					check_variable = { which = global_imp_auth value > 70 }
				}
				set_variable = { which = global_imp_auth value = 60 }
				set_character_flag = imp_auth_dont_change
			}
		}
	}
	
	option = { name = OK }
}
