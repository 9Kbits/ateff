# Safety catch - clears character flags and modifiers
character_event = {
	id = con.2

	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		remove_character_modifier = holding_congress
		clr_character_flag = do_not_disturb
		clr_character_flag = guest_congress_started
		remove_character_modifier = holding_congress
	}
}

#Send invites
character_event = {
	id = con.3
	desc = "EVTDESCCON3"
	picture = GFX_evt_feast

	is_triggered_only = yes

	only_rulers = yes
	capable_only = yes
	prisoner = no

	trigger = {
		war = no
		has_character_modifier = holding_congress
		NOT = {
			has_character_flag = sent_invitations
		}
	}
	
	option = {
		name = "EVTOPTA72010"
		set_character_flag = sent_invitations
		any_vassal = {
			limit = {
				prisoner = no
				NOT = { trait = incapable }
				NOT = { has_character_modifier = holding_congress }
				war = no
				is_adult = yes
			}
			letter_event = {
				id = con.4
				days = 1
				random = 2
				tooltip = "EVTTOOLTIP72011"
			}
		}
	}
}
