# Events for electing the President
# by Sam
namespace = president
# How it works:
# #1 is triggered from on_death. It makes the presidency vacant, calls #2, and primes #4.
# #2 makes sure the electoral college (used for processing instead of the Pope) doesn't die mid-election and fuck everything up, and calls #3 for every Americanist character, asking them if they want to run or not.
# #4, #5, and #6 implement a while loop. In #4, if there are more than 4 candidates, it calls #6 for every candidate (using #5 so FROM is properly set), which decides whether they should be eliminated or not, and then calls #4 again. If there are less than 5 candidates, it gives them traits indicating they're a candidate (set to cached = yes to allow quick scoping), calls #10 to inform players who the candidates are, and schedules #90 for the actual election.
# #90 calls #100 for every Americanist ruler and schedules #101. #100 is where each character casts their vote. #101 determines who won. The variable cXvY is the number of votes for candidate X minus the number of votes for candidate Y, so we check if it's positive or negative to see if one candidate has more or less votes than another.
# #101 calls #110 for the winner of the election, telling them they won and giving them the title, and calling #111 for all Americanist players to inform them of the winner.


# Send out notifications of President death
character_event = {
	id = president.1
	desc = "President is dead"
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		has_landed_title = d_americanist
	}
	
	immediate = {
		#d_americanist = { destroy_landed_title = THIS }
		d_electoral_college = { character_event = { id = president.115 } } # Clean up variables
		d_electoral_college = { holder_scope = { character_event = { id = president.2 days = 1 } } }
		d_electoral_college = { holder_scope = { character_event = { id = president.4 days = 2 } } }
	}
	
	option = {
		name = "Shouldn't see this"
		set_global_flag = pres_election
	}
}

character_event = {
	id = president.2
	desc = "President is dead"
	is_triggered_only = yes

	option = {
		name = "Shouldn't see this"
		d_electoral_college = { holder_scope = {

			health = 10

			set_variable = { which = "candidate1_votes" value = 0 }
			set_variable = { which = "candidate2_votes" value = 0 }
			set_variable = { which = "candidate3_votes" value = 0 }
			set_variable = { which = "candidate4_votes" value = 0 }
			set_variable = { which = "c1v2" value = 0 }
			set_variable = { which = "c1v3" value = 0 }
			set_variable = { which = "c1v4" value = 0 }
			set_variable = { which = "c2v3" value = 0 }
			set_variable = { which = "c2v4" value = 0 }
			set_variable = { which = "c3v4" value = 0 }
		} }
		any_independent_ruler = {
			any_realm_character = {
				limit = {
					religion = americanist
					is_adult = yes
					#higher_tier_than = COUNT # Dukes and above
				}
				character_event = { id = president.3 }
			}
		}
	}
}

					NOT = { has_trait = pres_candidate_2 }

			}
		}
		random_independent_ruler = {
			limit = {
				any_realm_character = {
					religion = americanist
					has_character_flag = running_for_president
					NOT = { has_trait = pres_candidate_1 }
					NOT = { has_trait = pres_candidate_2 }
					NOT = { has_trait = pres_candidate_3 }
					NOT = { has_trait = pres_candidate_4 }
				}

				limit = {
					religion = americanist
					has_character_flag = running_for_president
					NOT = { has_trait = pres_candidate_1 }
					NOT = { has_trait = pres_candidate_2 }
					NOT = { has_trait = pres_candidate_3 }
					NOT = { has_trait = pres_candidate_4 }
				}
				add_trait = pres_candidate_3
				#set_character_flag = pres_candidate_3
